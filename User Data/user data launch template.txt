#!/bin/bash
sudo -i
yum update -y
yum install -y httpd
systemctl start httpd
systemctl enable httpd
yum install php php-mysqli -y
yum install mariadb105-server -y
systemctl start mariadb
systemctl enable mariadb
systemctl restart httpd
rm -rf /var/www/html
cd /var/www/
mkdir html
mount -t nfs4 -o nfsvers=4.1,rsize=1048576,wsize=1048576,hard,timeo=600,retrans=2,noresvport fs-0b26b24bc90fc3fe9.efs.us-east-1.amazonaws.com:/ /var/www/html
echo "fs-0b26b24bc90fc3fe9.efs.us-east-1.amazonaws.com:/ /var/www/html nfs4 nfsvers=4.1,rsize=1048576,wsize=1048576,hard,timeo=600,retrans=2,noresvport,_netdev 0 0" >> /etc/fstab

Copy NFS attach with DNS and point to /var/www/html
